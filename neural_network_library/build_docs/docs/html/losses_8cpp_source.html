21
22 // MSE is implemented compositionally so existing tensor autograd rules can

29Tensor binary_cross_entropy(const Tensor& predictions, const Tensor& targets) {
30 if (predictions.shape() != targets.shape()) {
31 throw std::invalid_argument("Predictions and targets must have the same shape");
32 }
33
34 // Clamp probabilities slightly away from 0 and 1 to avoid log singularities.
35 const double epsilon = 1e-7;
36 auto dloss_dpred = std::make_shared<std::vector<double>>(predictions.size(), 0.0);
37 double total_loss = 0.0;
38
39 for (size_t i = 0; i < predictions.size(); ++i) {
40 double p = std::max(epsilon, std::min(1.0 - epsilon, predictions[i]));
41 double y = targets[i];
42 const double sample_loss = -(y * std::log(p) + (1.0 - y) * std::log(1.0 - p));
43 total_loss += sample_loss;
44 // Precompute the per-element derivative so backward only needs to scale
45 // it by the upstream scalar gradient.
46 (*dloss_dpred)[i] = (-((y / p) - ((1.0 - y) / (1.0 - p)))) /
47 static_cast<double>(predictions.size());
48 }
49
50 const double mean_loss = total_loss / static_cast<double>(predictions.size());
51 Tensor result({mean_loss}, {1}, predictions.requires_grad());
52
53 if (result.requires_grad()) {
54 auto pred_ptr = Tensor::alias(predictions);
55 auto out_grad = result.grad_ptr();
56 result.set_autograd(
57 [pred_ptr, out_grad, dloss_dpred]() {
58 if (!out_grad || !pred_ptr->requires_grad()) {
59 return;
60 }
61 pred_ptr->ensure_grad();
62 for (size_t i = 0; i < pred_ptr->size(); ++i) {
63 // The loss is scalar, so out_grad[0] is the upstream scale.
64 pred_ptr->grad()[i] += (*out_grad)[0] * (*dloss_dpred)[i];
65 }
66 },
67 {pred_ptr}
68 );
69 }
70
71 return result;
72}

74Tensor cross_entropy_loss(const Tensor& logits, const Tensor& targets) {
75 if (logits.shape().size() != 2 || targets.shape().size() != 2) {
76 throw std::invalid_argument("Cross entropy expects 2D tensors (batch_size x num_classes)");
77 }
78
79 if (logits.shape()[0] != targets.shape()[0]) {
80 throw std::invalid_argument("Batch sizes must match");
81 }
82
83 // Convert logits to probabilities before computing the negative log-likelihood.
84 Tensor probs = activations::softmax(logits);
85
86 const double epsilon = 1e-7;
87 size_t batch_size = logits.shape()[0];
88 size_t num_classes = logits.shape()[1];
89
90 double total_loss = 0.0;
91
92 for (size_t i = 0; i < batch_size; ++i) {
93 for (size_t j = 0; j < num_classes; ++j) {
94 size_t idx = i * num_classes + j;
95 // For one-hot targets only the true class contributes to the loss.
96 if (targets[idx] > 0.0) {
97 double p = std::max(epsilon, probs[idx]);
98 total_loss -= targets[idx] * std::log(p);
99 }
100 }
101 }
102
103 total_loss /= static_cast<double>(batch_size);
104
105 return Tensor({total_loss}, {1}, logits.requires_grad());
106}

108Tensor mae_loss(const Tensor& predictions, const Tensor& targets) {
109 if (predictions.shape() != targets.shape()) {
110 throw std::invalid_argument("Predictions and targets must have the same shape");
111 }
112
113 Tensor diff = predictions - targets;
114 Tensor abs_diff(diff.shape(), diff.requires_grad());
115
116 for (size_t i = 0; i < diff.size(); ++i) {
117 // Store the absolute residual before averaging.
118 abs_diff[i] = std::abs(diff[i]);
119 }
120
121 return abs_diff.mean();
122}
