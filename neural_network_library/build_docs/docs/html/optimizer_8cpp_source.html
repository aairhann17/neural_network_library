46void SGD::step() {
47 for (auto* param : parameters_) {
48 if (!param->has_grad()) {
49 continue;
50 }
51
52 const Tensor& grad = param->grad();
53
54 // Apply weight decay (L2 regularization)
55 Tensor effective_grad = grad;
56 if (weight_decay_ > 0.0) {
57 // Weight decay adds an L2 penalty term directly to the gradient.
58 for (size_t i = 0; i < param->size(); ++i) {
59 effective_grad[i] += weight_decay_ * (*param)[i];
60 }
61 }
62
63 if (momentum_ > 0.0) {
64 // Velocity update: v = momentum * v + grad
65 auto velocity_it = velocities_.find(param);
66 if (velocity_it == velocities_.end()) {
67 velocity_it = velocities_.emplace(param, Tensor::zeros(param->shape())).first;
68 }
69 Tensor& velocity = velocity_it->second;
70 for (size_t i = 0; i < velocity.size(); ++i) {
71 // Exponentially smooth recent gradients into the velocity buffer.
72 velocity[i] = momentum_ * velocity[i] + effective_grad[i];
73 }
74
75 // Parameter update: param -= learning_rate * velocity
76 for (size_t i = 0; i < param->size(); ++i) {
77 (*param)[i] -= learning_rate_ * velocity[i];
78 }
79 } else {
80 // Standard SGD: param -= learning_rate * grad
81 for (size_t i = 0; i < param->size(); ++i) {
82 (*param)[i] -= learning_rate_ * effective_grad[i];
83 }
84 }
85 }
86}

109void Adam::step() {
110 // Bias correction depends on the number of update steps already taken.
111 t_++;
112
113 for (auto* param : parameters_) {
114 if (!param->has_grad()) {
115 continue;
116 }
117
118 const Tensor& grad = param->grad();
119
120 // Apply weight decay (L2 regularization)
121 Tensor effective_grad = grad;
122 if (weight_decay_ > 0.0) {
123 for (size_t i = 0; i < param->size(); ++i) {
124 effective_grad[i] += weight_decay_ * (*param)[i];
125 }
126 }
127
128 auto m_it = m_.find(param);
129 if (m_it == m_.end()) {
130 m_it = m_.emplace(param, Tensor::zeros(param->shape())).first;
131 }
132 auto v_it = v_.find(param);
133 if (v_it == v_.end()) {
134 v_it = v_.emplace(param, Tensor::zeros(param->shape())).first;
135 }
136
137 Tensor& m = m_it->second; // first moment
138 Tensor& v = v_it->second; // second moment
139
140 // Update first and second moments, then convert them to bias-corrected
141 // estimates before applying the parameter update.
142 for (size_t i = 0; i < param->size(); ++i) {
143 m[i] = beta1_ * m[i] + (1.0 - beta1_) * effective_grad[i];
144
145 v[i] = beta2_ * v[i] + (1.0 - beta2_) * effective_grad[i] * effective_grad[i];
146
147 double m_hat = m[i] / (1.0 - std::pow(beta1_, t_));
148 double v_hat = v[i] / (1.0 - std::pow(beta2_, t_));
149
150 // Scale by the normalized second moment so large-gradient dimensions
151 // receive smaller effective step sizes.
152 (*param)[i] -= learning_rate_ * m_hat / (std::sqrt(v_hat) + epsilon_);
153 }
154 }
155}
