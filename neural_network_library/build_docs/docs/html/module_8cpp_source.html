36void Linear::initialize_parameters() {
37 // Xavier/Glorot scaling keeps activation magnitudes in a reasonable range
38 // for fully connected networks at initialization time.
39 double std = std::sqrt(2.0 / static_cast<double>(in_features_ + out_features_));
40
41 weight_ = Tensor::randn({out_features_, in_features_}, true);
42 weight_ = weight_ * std;
43
44 if (use_bias_) {
45 // A zero bias is a common default and keeps the layer centered at start.
46 bias_ = Tensor::zeros({out_features_}, true);
47 }
48}

50Tensor Linear::forward(const Tensor& input) {
51 // input shape: (batch_size, in_features)
52 // weight shape: (out_features, in_features)
53 // output shape: (batch_size, out_features)
54
55 if (input.shape().size() != 2) {
56 throw std::invalid_argument("Linear layer expects 2D input (batch_size, in_features)");
57 }
58
59 if (input.shape()[1] != in_features_) {
60 throw std::invalid_argument(
61 "Input features mismatch. Expected: " + std::to_string(in_features_) +
62 ", got: " + std::to_string(input.shape()[1])
63 );
64 }
65
66 // Transpose the stored weight matrix so each row in the batch is multiplied
67 // by one output neuron weight vector.
68 Tensor output = input.matmul(weight_.transpose());
69
70 // Bias is stored as a 1D vector, so add it manually across every batch row.
71 if (use_bias_) {
72 for (size_t i = 0; i < output.shape()[0]; ++i) {
73 for (size_t j = 0; j < output.shape()[1]; ++j) {
74 output[i * output.shape()[1] + j] += bias_[j];
75 }
76 }
77 }
78
79 return output;
80}

115Tensor Dropout::forward(const Tensor& input) {
116 if (!is_training()) {
117 // During evaluation dropout becomes a no-op so predictions stay stable.
118 return input;
119 }
120
121 Tensor output(input.shape(), input.requires_grad());
122
123 std::random_device rd;
124 std::mt19937 gen(rd());
125 std::bernoulli_distribution dist(1.0 - p_);
126
127 // Inverted dropout scales surviving activations during training so no extra
128 // rescaling is required at inference time.
129 double scale = 1.0 / (1.0 - p_);
130
131 for (size_t i = 0; i < input.size(); ++i) {
132 if (dist(gen)) {
133 output[i] = input[i] * scale;
134 } else {
135 // Masked activations contribute nothing for this forward pass.
136 output[i] = 0.0;
137 }
138 }
139
140 return output;
141}
