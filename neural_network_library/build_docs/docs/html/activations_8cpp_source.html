17Tensor relu(const Tensor& input) {
18 Tensor output(input.shape(), input.requires_grad());
19 // Keep a binary mask so backward can cheaply gate gradients.
20 auto mask = std::make_shared<std::vector<double>>(input.size(), 0.0);
21
22 for (size_t i = 0; i < input.size(); ++i) {
23 if (input[i] > 0.0) {
24 output[i] = input[i];
25 (*mask)[i] = 1.0;
26 } else {
27 output[i] = 0.0;
28 }
29 }
30
31 if (output.requires_grad()) {
32 auto input_ptr = Tensor::alias(input);
33 auto out_grad = output.grad_ptr();
34 output.set_autograd(
35 [input_ptr, out_grad, mask]() {
36 if (!out_grad || !input_ptr->requires_grad()) {
37 return;
38 }
39 input_ptr->ensure_grad();
40 for (size_t i = 0; i < input_ptr->size(); ++i) {
41 // ReLU only passes gradients through positive inputs.
42 input_ptr->grad()[i] += (*out_grad)[i] * (*mask)[i];
43 }
44 },
45 {input_ptr}
46 );
47 }
48
49 return output;
50}

52Tensor sigmoid(const Tensor& input) {
53 Tensor output(input.shape(), input.requires_grad());
54 // Cache sigmoid outputs because the derivative depends on s * (1 - s).
55 auto sigmoid_values = std::make_shared<std::vector<double>>(input.size(), 0.0);
56
57 for (size_t i = 0; i < input.size(); ++i) {
58 const double s = 1.0 / (1.0 + std::exp(-input[i]));
59 output[i] = s;
60 (*sigmoid_values)[i] = s;
61 }
62
63 if (output.requires_grad()) {
64 auto input_ptr = Tensor::alias(input);
65 auto out_grad = output.grad_ptr();
66 output.set_autograd(
67 [input_ptr, out_grad, sigmoid_values]() {
68 if (!out_grad || !input_ptr->requires_grad()) {
69 return;
70 }
71 input_ptr->ensure_grad();
72 for (size_t i = 0; i < input_ptr->size(); ++i) {
73 const double s = (*sigmoid_values)[i];
74 // d/dx sigmoid(x) = sigmoid(x) * (1 - sigmoid(x)).
75 input_ptr->grad()[i] += (*out_grad)[i] * s * (1.0 - s);
76 }
77 },
78 {input_ptr}
79 );
80 }
81
82 return output;
83}

96Tensor softmax(const Tensor& input, int axis) {
97 if (input.shape().size() != 2) {
98 throw std::invalid_argument("Softmax currently only supports 2D tensors");
99 }
100
101 // This implementation normalizes across the class dimension of each row.
102 // The axis argument is accepted for API compatibility but only row-wise
103 // softmax is implemented today.
104 (void)axis;
105 size_t rows = input.shape()[0];
106 size_t cols = input.shape()[1];
107
108 Tensor output(input.shape(), input.requires_grad());
109
110 for (size_t i = 0; i < rows; ++i) {
111 // Find max for numerical stability
112 double max_val = input[i * cols];
113 for (size_t j = 1; j < cols; ++j) {
114 max_val = std::max(max_val, input[i * cols + j]);
115 }
116
117 // Compute exp and sum
118 double sum_exp = 0.0;
119 for (size_t j = 0; j < cols; ++j) {
120 double exp_val = std::exp(input[i * cols + j] - max_val);
121 output[i * cols + j] = exp_val;
122 sum_exp += exp_val;
123 }
124
125 // Normalize
126 for (size_t j = 0; j < cols; ++j) {
127 output[i * cols + j] /= sum_exp;
128 }
129 }
130
131 return output;
132}

134Tensor leaky_relu(const Tensor& input, double alpha) {
135 Tensor output(input.shape(), input.requires_grad());
136
137 for (size_t i = 0; i < input.size(); ++i) {
138 // Preserve a small slope for negative inputs to avoid dead activations.
139 output[i] = input[i] > 0.0 ? input[i] : alpha * input[i];
140 }
141
142 return output;
143}

145Tensor elu(const Tensor& input, double alpha) {
146 Tensor output(input.shape(), input.requires_grad());
147
148 for (size_t i = 0; i < input.size(); ++i) {
149 // ELU is linear for positive inputs and smoothly saturating below zero.
150 output[i] = input[i] > 0.0 ? input[i] : alpha * (std::exp(input[i]) - 1.0);
151 }
152
153 return output;
